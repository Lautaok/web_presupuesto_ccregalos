    <script>
        // Función para formatear números como moneda
        function formatCurrency(value) {
            return new Intl.NumberFormat('es-AR', {
                style: 'currency',
                currency: 'ARS',
                minimumFractionDigits: 2,
                maximumFractionDigits: 2
            }).format(value);
        }

        // Función principal para calcular todo
        function calcularPresupuesto() {
            // --- 1. Obtener valores de los inputs ---
            const costoTintaSet = parseFloat(document.getElementById('costoTintaSet').value) || 0;
            const rendimientoTinta = parseFloat(document.getElementById('rendimientoTinta').value) || 1;
            const costoPapelPaquete = parseFloat(document.getElementById('costoPapelPaquete').value) || 0;
            const unidadesPapel = parseFloat(document.getElementById('unidadesPapel').value) || 1;
            const costoDiseno = parseFloat(document.getElementById('costoDiseno').value) || 0;
            const margenGanancia = parseFloat(document.getElementById('margenGanancia').value) || 0;
            const cantidadHojas = parseFloat(document.getElementById('cantidadHojas').value) || 0;
            const incluyeDiseno = document.getElementById('incluyeDiseno').value;

            // --- 2. Cálculos de Configuración ---
            const costoTintaPorPagina = costoTintaSet / rendimientoTinta;
            const costoPorHoja = costoPapelPaquete / unidadesPapel;

            // --- 3. Cálculos del Desglose ---
            const totalCostoTinta = costoTintaPorPagina * cantidadHojas;
            const totalCostoPapel = costoPorHoja * cantidadHojas;
            const totalCostoDiseno = (incluyeDiseno === 'si') ? costoDiseno : 0;
            
            const costoTotalProduccion = totalCostoTinta + totalCostoPapel + totalCostoDiseno;

            // --- 4. Cálculos del Precio de Venta ---
            const ganancia = costoTotalProduccion * (margenGanancia / 100);
            const precioFinal = costoTotalProduccion + ganancia;
            
            // --- 5. Cálculo del Dato Clave ---
            const montoSeparar = totalCostoTinta + totalCostoPapel;

            // --- 6. Actualizar la interfaz con los resultados formateados ---
            document.getElementById('costoTintaPorPagina').textContent = formatCurrency(costoTintaPorPagina);
            document.getElementById('costoPorHoja').textContent = formatCurrency(costoPorHoja);
            
            document.getElementById('totalCostoTinta').textContent = formatCurrency(totalCostoTinta);
            document.getElementById('totalCostoPapel').textContent = formatCurrency(totalCostoPapel);
            document.getElementById('totalCostoDiseno').textContent = formatCurrency(totalCostoDiseno);
            document.getElementById('costoTotalProduccion').textContent = formatCurrency(costoTotalProduccion);
            
            document.getElementById('ganancia').textContent = formatCurrency(ganancia);
            document.getElementById('precioFinal').textContent = formatCurrency(precioFinal);
            
            document.getElementById('montoSeparar').textContent = formatCurrency(montoSeparar);
        }

        // --- Event Listeners ---
        // Seleccionamos todos los inputs y el select
        const inputs = document.querySelectorAll('input[type="number"], input[type="text"], select');
        
        // Añadimos un listener a cada uno que llame a la función de cálculo
        inputs.forEach(input => {
            input.addEventListener('input', calcularPresupuesto);
        });

        // Calculamos todo una vez al cargar la página con los valores por defecto
        document.addEventListener('DOMContentLoaded', calcularPresupuesto);
    </script>
